下载解压至c盘根目录，管理员命令窗

http://www.apachehaus.com/cgi-bin/download.plx

cd \Apache24
cd bin
httpd -t
httpd -k install -n Apache



引用httpd-vhosts

C:\Apache24\conf\httpd.conf

# Virtual hosts
Include conf/extra/httpd-vhosts.conf



分配目录域名,权限

C:\Apache24\conf\extra\httpd-vhosts.conf

# site_1
<virtualhost *:80>
    DocumentRoot "D:\Dropbox\local"
    ServerName local
</virtualhost>
NameVirtualHost 127.0.0.2:80
<directory d:\dropbox\local>
    AllowOverride All
    Require all granted
</directory>
 
# site_2
<virtualhost *:80>
    DocumentRoot "D:\java\API\java_platform_se_8"
    ServerName java_platform_se_8
</virtualhost>
NameVirtualHost 127.0.0.3:80
<directory d:\java\api\java_platform_se_8>
    AllowOverride All
    Require all granted
</directory>
 
# site_3
<virtualhost *:80>
    DocumentRoot "D:\java\API\java_platform_se_11"
    ServerName java_platform_se_11
</virtualhost>
NameVirtualHost 127.0.0.4:80
<directory d:\java\java_platform_se_11>
    AllowOverride All
    Require all granted
</directory>



写入hosts

C:\Windows\System32\drivers\etc\hosts
